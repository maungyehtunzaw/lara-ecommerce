# Objective

# Installation
 - clone the repository.
 - composer install
 - npm install
 - cp .env.example .env (or) copy .env.example and rename .env
 - php artisan key:generate
 - php artisan migrate (or) import datbases/db.sql 
 - php artisan db:seed
 - php artisan passport:client --personal
 - php artisan ser
 - php artisan passport
